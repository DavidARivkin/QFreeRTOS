117 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters )
118 {
119 StackType_t DS_Reg = 0;
120 
121  /* We need the true data segment. */
122  __asm{ MOV DS_Reg, DS };
123 
124  /* Place a few bytes of known values on the bottom of the stack.
125  This is just useful for debugging. */
126 
127  *pxTopOfStack = 0x1111;
128  pxTopOfStack--;
129  *pxTopOfStack = 0x2222;
130  pxTopOfStack--;
131  *pxTopOfStack = 0x3333;
132  pxTopOfStack--;
133 
134  /* We are going to start the scheduler using a return from interrupt
135  instruction to load the program counter, so first there would be the
136  function call with parameters preamble. */
137 
138  *pxTopOfStack = FP_OFF( pvParameters );
139  pxTopOfStack--;
140  *pxTopOfStack = FP_OFF( pxCode );
141  pxTopOfStack--;
142 
143  /* Next the status register and interrupt return address. */
144  *pxTopOfStack = portINITIAL_SW;
145  pxTopOfStack--;
146  *pxTopOfStack = FP_SEG( pxCode );
147  pxTopOfStack--;
148  *pxTopOfStack = FP_OFF( pxCode );
149  pxTopOfStack--;
150 
151  /* The remaining registers would be pushed on the stack by our context
152  switch function. These are loaded with values simply to make debugging
153  easier. */
154  *pxTopOfStack = ( StackType_t ) 0xAAAA; /* AX */
155  pxTopOfStack--;
156  *pxTopOfStack = ( StackType_t ) 0xBBBB; /* BX */
157  pxTopOfStack--;
158  *pxTopOfStack = ( StackType_t ) 0xCCCC; /* CX */
159  pxTopOfStack--;
160  *pxTopOfStack = ( StackType_t ) 0xDDDD; /* DX */
161  pxTopOfStack--;
162  *pxTopOfStack = ( StackType_t ) 0xEEEE; /* ES */
163  pxTopOfStack--;
164 
165  *pxTopOfStack = DS_Reg; /* DS */
166  pxTopOfStack--;
167  *pxTopOfStack = ( StackType_t ) 0x0123; /* SI */
168  pxTopOfStack--;
169  *pxTopOfStack = ( StackType_t ) 0xDDDD; /* DI */
170  pxTopOfStack--;
171  *pxTopOfStack = ( StackType_t ) 0xBBBB; /* BP */
172 
173  return pxTopOfStack;
174 }

177 BaseType_t xPortStartScheduler( void )
178 {
179  /* This is called with interrupts already disabled. */
180 
181  /* Put our manual switch (yield) function on a known
182  vector. */
183  setvect( portSWITCH_INT_NUMBER, prvYieldProcessor );
184 
185  /* Setup the tick interrupt. */
186  prvSetupTimerInterrupt();
187 
188  /* Kick off the scheduler by setting up the context of the first task. */
189  portFIRST_CONTEXT();
190 
191  /* Should not get here! */
192  return pdFALSE;
193 }

196 /* The ISR used depends on whether the preemptive or cooperative scheduler
197 is being used. */
198 #if( configUSE_PREEMPTION == 1 )
199  static void __interrupt __far prvPreemptiveTick( void )
200  {
201  /* Get the scheduler to update the task states following the tick. */
202  if( xTaskIncrementTick() != pdFALSE )
203  {
204  /* Switch in the context of the next task to be run. */
205  portEND_SWITCHING_ISR();
206  }
207 
208  /* Reset interrupt. */
209  outport( portEIO_REGISTER, portCLEAR_INTERRUPT );
210  }
211 #else
212  static void __interrupt __far prvNonPreemptiveTick( void )
213  {
214  /* Same as preemptive tick, but the cooperative scheduler is being used
215  so we don't have to switch in the context of the next task. */
216  xTaskIncrementTick();
217 
218  /* Reset interrupt. */
219  outport( portEIO_REGISTER, portCLEAR_INTERRUPT );
220  }
221 #endif
222 /*-----------------------------------------------------------*/
223 
224 static void __interrupt __far prvYieldProcessor( void )
225 {
226  /* Switch in the context of the next task to be run. */
227  portEND_SWITCHING_ISR();
228 }

237 static void prvSetupTimerInterrupt( void )
238 {
239 const uint32_t ulCompareValue = portTIMER_COMPARE;
240 uint16_t usTimerCompare;
241 
242  usTimerCompare = ( uint16_t ) ( ulCompareValue >> 4 );
243  t2_init( portENABLE_TIMER, portPRESCALE_VALUE, NULL );
244 
245  #if( configUSE_PREEMPTION == 1 )
246  /* Tick service routine used by the scheduler when preemptive scheduling is
247  being used. */
248  t1_init( portENABLE_TIMER_AND_INTERRUPT, usTimerCompare, usTimerCompare, prvPreemptiveTick );
249  #else
250  /* Tick service routine used by the scheduler when cooperative scheduling is
251  being used. */
252  t1_init( portENABLE_TIMER_AND_INTERRUPT, usTimerCompare, usTimerCompare, prvNonPreemptiveTick );
253  #endif
254 }
