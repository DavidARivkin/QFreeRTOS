134 /*
135  * Initialise the stack of a task to look exactly as if a call to
136  * portSAVE_CONTEXT had been called.
137  *
138  * See header file for description.
139  */
140 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters )
141 {
142 StackType_t *pxOriginalTOS;
143 
144  pxOriginalTOS = pxTopOfStack;
145 
146  /* Setup the initial stack of the task. The stack is set exactly as
147  expected by the portRESTORE_CONTEXT() macro. */
148 
149  /* First on the stack is the return address - which in this case is the
150  start of the task. The offset is added to make the return address appear
151  as it would within an IRQ ISR. */
152  *pxTopOfStack = ( StackType_t ) pxCode + portINSTRUCTION_SIZE;
153  pxTopOfStack--;
154 
155  *pxTopOfStack = ( StackType_t ) 0xaaaaaaaa; /* R14 */
156  pxTopOfStack--;
157  *pxTopOfStack = ( StackType_t ) pxOriginalTOS; /* Stack used when task starts goes in R13. */
158  pxTopOfStack--;
159  *pxTopOfStack = ( StackType_t ) 0x12121212; /* R12 */
160  pxTopOfStack--;
161  *pxTopOfStack = ( StackType_t ) 0x11111111; /* R11 */
162  pxTopOfStack--;
163  *pxTopOfStack = ( StackType_t ) 0x10101010; /* R10 */
164  pxTopOfStack--;
165  *pxTopOfStack = ( StackType_t ) 0x09090909; /* R9 */
166  pxTopOfStack--;
167  *pxTopOfStack = ( StackType_t ) 0x08080808; /* R8 */
168  pxTopOfStack--;
169  *pxTopOfStack = ( StackType_t ) 0x07070707; /* R7 */
170  pxTopOfStack--;
171  *pxTopOfStack = ( StackType_t ) 0x06060606; /* R6 */
172  pxTopOfStack--;
173  *pxTopOfStack = ( StackType_t ) 0x05050505; /* R5 */
174  pxTopOfStack--;
175  *pxTopOfStack = ( StackType_t ) 0x04040404; /* R4 */
176  pxTopOfStack--;
177  *pxTopOfStack = ( StackType_t ) 0x03030303; /* R3 */
178  pxTopOfStack--;
179  *pxTopOfStack = ( StackType_t ) 0x02020202; /* R2 */
180  pxTopOfStack--;
181  *pxTopOfStack = ( StackType_t ) 0x01010101; /* R1 */
182  pxTopOfStack--;
183 
184  /* When the task starts is will expect to find the function parameter in
185  R0. */
186  *pxTopOfStack = ( StackType_t ) pvParameters; /* R0 */
187  pxTopOfStack--;
188 
189  /* The status register is set for system mode, with interrupts enabled. */
190  *pxTopOfStack = ( StackType_t ) portINITIAL_SPSR;
191 
192  if( ( ( uint32_t ) pxCode & 0x01UL ) != 0x00UL )
193  {
194  /* We want the task to start in thumb mode. */
195  *pxTopOfStack |= portTHUMB_MODE_BIT;
196  }
197 
198  pxTopOfStack--;
199 
200  /* Interrupt flags cannot always be stored on the stack and will
201  instead be stored in a variable, which is then saved as part of the
202  tasks context. */
203  *pxTopOfStack = portNO_CRITICAL_NESTING;
204 
205  return pxTopOfStack;
206 }

274 static void prvSetupTimerInterrupt( void )
275 {
276 uint32_t ulCompareMatch;
277 
278  /* A 1ms tick does not require the use of the timer prescale. This is
279  defaulted to zero but can be used if necessary. */
280  T0PR = portPRESCALE_VALUE;
281 
282  /* Calculate the match value required for our wanted tick rate. */
283  ulCompareMatch = configCPU_CLOCK_HZ / configTICK_RATE_HZ;
284 
285  /* Protect against divide by zero. Using an if() statement still results
286  in a warning - hence the #if. */
287  #if portPRESCALE_VALUE != 0
288  {
289  ulCompareMatch /= ( portPRESCALE_VALUE + 1 );
290  }
291  #endif
292 
293  T0MR0 = ulCompareMatch;
294 
295  /* Generate tick with timer 0 compare match. */
296  T0MCR = portRESET_COUNT_ON_MATCH | portINTERRUPT_ON_MATCH;
297 
298  /* Setup the VIC for the timer. */
299  VICIntSelect &= ~( portTIMER_VIC_CHANNEL_BIT );
300  VICIntEnable |= portTIMER_VIC_CHANNEL_BIT;
301 
302  /* The ISR installed depends on whether the preemptive or cooperative
303  scheduler is being used. */
304  #if configUSE_PREEMPTION == 1
305  {
306  extern void ( vPortPreemptiveTickEntry )( void );
307 
308  VICVectAddr0 = ( uint32_t ) vPortPreemptiveTickEntry;
309  }
310  #else
311  {
312  extern void ( vNonPreemptiveTick )( void );
313 
314  VICVectAddr0 = ( int32_t ) vPortNonPreemptiveTick;
315  }
316  #endif
317 
318  VICVectCntl0 = portTIMER_VIC_CHANNEL | portTIMER_VIC_ENABLE;
319 
320  /* Start the timer - interrupts are disabled when this function is called
321  so it is okay to do this here. */
322  T0TCR = portENABLE_TIMER;
323 }

326 void vPortEnterCritical( void )
327 {
328  /* Disable interrupts first! */
329  __disable_interrupt();
330 
331  /* Now interrupts are disabled ulCriticalNesting can be accessed
332  directly. Increment ulCriticalNesting to keep a count of how many times
333  portENTER_CRITICAL() has been called. */
334  ulCriticalNesting++;
335 }
336 /*-----------------------------------------------------------*/
337 
338 void vPortExitCritical( void )
339 {
340  if( ulCriticalNesting > portNO_CRITICAL_NESTING )
341  {
342  /* Decrement the nesting count as we are leaving a critical section. */
343  ulCriticalNesting--;
344 
345  /* If the nesting level has reached zero then interrupts should be
346  re-enabled. */
347  if( ulCriticalNesting == portNO_CRITICAL_NESTING )
348  {
349  __enable_interrupt();
350  }
351  }
352 }
