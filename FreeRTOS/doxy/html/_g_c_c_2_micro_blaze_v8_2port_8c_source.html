126 /* If an interrupt requests a context switch, then ulTaskSwitchRequested will
127 get set to 1. ulTaskSwitchRequested is inspected just before the main interrupt
128 handler exits. If, at that time, ulTaskSwitchRequested is set to 1, the kernel

131 an unusual mechanism, but is used for this port because a single interrupt can

148 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters )
149 {
150 extern void *_SDA2_BASE_, *_SDA_BASE_;
151 const uint32_t ulR2 = ( uint32_t ) &_SDA2_BASE_;
152 const uint32_t ulR13 = ( uint32_t ) &_SDA_BASE_;
153 
154  /* Place a few bytes of known values on the bottom of the stack.
155  This is essential for the Microblaze port and these lines must
156  not be omitted. */
157  *pxTopOfStack = ( StackType_t ) 0x00000000;
158  pxTopOfStack--;
159  *pxTopOfStack = ( StackType_t ) 0x00000000;
160  pxTopOfStack--;
161  *pxTopOfStack = ( StackType_t ) 0x00000000;
162  pxTopOfStack--;
163 
164  #if( XPAR_MICROBLAZE_USE_FPU != 0 )
165  /* The FSR value placed in the initial task context is just 0. */
166  *pxTopOfStack = portINITIAL_FSR;
167  pxTopOfStack--;
168  #endif
169 
170  /* The MSR value placed in the initial task context should have interrupts
171  disabled. Each task will enable interrupts automatically when it enters
172  the running state for the first time. */
173  *pxTopOfStack = mfmsr() & ~portMSR_IE;
174 
175  #if( MICROBLAZE_EXCEPTIONS_ENABLED == 1 )
176  {
177  /* Ensure exceptions are enabled for the task. */
178  *pxTopOfStack |= portMSR_EE;
179  }
180  #endif
181 
182  pxTopOfStack--;
183 
184  /* First stack an initial value for the critical section nesting. This
185  is initialised to zero. */
186  *pxTopOfStack = ( StackType_t ) 0x00;
187 
188  /* R0 is always zero. */
189  /* R1 is the SP. */
190 
191  /* Place an initial value for all the general purpose registers. */
192  pxTopOfStack--;
193  *pxTopOfStack = ( StackType_t ) ulR2; /* R2 - read only small data area. */
194  pxTopOfStack--;
195  *pxTopOfStack = ( StackType_t ) 0x03; /* R3 - return values and temporaries. */
196  pxTopOfStack--;
197  *pxTopOfStack = ( StackType_t ) 0x04; /* R4 - return values and temporaries. */
198  pxTopOfStack--;
199  *pxTopOfStack = ( StackType_t ) pvParameters;/* R5 contains the function call parameters. */
200 
201  #ifdef portPRE_LOAD_STACK_FOR_DEBUGGING
202  pxTopOfStack--;
203  *pxTopOfStack = ( StackType_t ) 0x06; /* R6 - other parameters and temporaries. Used as the return address from vPortTaskEntryPoint. */
204  pxTopOfStack--;
205  *pxTopOfStack = ( StackType_t ) 0x07; /* R7 - other parameters and temporaries. */
206  pxTopOfStack--;
207  *pxTopOfStack = ( StackType_t ) 0x08; /* R8 - other parameters and temporaries. */
208  pxTopOfStack--;
209  *pxTopOfStack = ( StackType_t ) 0x09; /* R9 - other parameters and temporaries. */
210  pxTopOfStack--;
211  *pxTopOfStack = ( StackType_t ) 0x0a; /* R10 - other parameters and temporaries. */
212  pxTopOfStack--;
213  *pxTopOfStack = ( StackType_t ) 0x0b; /* R11 - temporaries. */
214  pxTopOfStack--;
215  *pxTopOfStack = ( StackType_t ) 0x0c; /* R12 - temporaries. */
216  pxTopOfStack--;
217  #else
218  pxTopOfStack-= 8;
219  #endif
220 
221  *pxTopOfStack = ( StackType_t ) ulR13; /* R13 - read/write small data area. */
222  pxTopOfStack--;
223  *pxTopOfStack = ( StackType_t ) pxCode; /* R14 - return address for interrupt. */
224  pxTopOfStack--;
225  *pxTopOfStack = ( StackType_t ) NULL; /* R15 - return address for subroutine. */
226 
227  #ifdef portPRE_LOAD_STACK_FOR_DEBUGGING
228  pxTopOfStack--;
229  *pxTopOfStack = ( StackType_t ) 0x10; /* R16 - return address for trap (debugger). */
230  pxTopOfStack--;
231  *pxTopOfStack = ( StackType_t ) 0x11; /* R17 - return address for exceptions, if configured. */
232  pxTopOfStack--;
233  *pxTopOfStack = ( StackType_t ) 0x12; /* R18 - reserved for assembler and compiler temporaries. */
234  pxTopOfStack--;
235  #else
236  pxTopOfStack -= 4;
237  #endif
238 
239  *pxTopOfStack = ( StackType_t ) 0x00; /* R19 - must be saved across function calls. Callee-save. Seems to be interpreted as the frame pointer. */
240 
241  #ifdef portPRE_LOAD_STACK_FOR_DEBUGGING
242  pxTopOfStack--;
243  *pxTopOfStack = ( StackType_t ) 0x14; /* R20 - reserved for storing a pointer to the Global Offset Table (GOT) in Position Independent Code (PIC). Non-volatile in non-PIC code. Must be saved across function calls. Callee-save. Not used by FreeRTOS. */
244  pxTopOfStack--;
245  *pxTopOfStack = ( StackType_t ) 0x15; /* R21 - must be saved across function calls. Callee-save. */
246  pxTopOfStack--;
247  *pxTopOfStack = ( StackType_t ) 0x16; /* R22 - must be saved across function calls. Callee-save. */
248  pxTopOfStack--;
249  *pxTopOfStack = ( StackType_t ) 0x17; /* R23 - must be saved across function calls. Callee-save. */
250  pxTopOfStack--;
251  *pxTopOfStack = ( StackType_t ) 0x18; /* R24 - must be saved across function calls. Callee-save. */
252  pxTopOfStack--;
253  *pxTopOfStack = ( StackType_t ) 0x19; /* R25 - must be saved across function calls. Callee-save. */
254  pxTopOfStack--;
255  *pxTopOfStack = ( StackType_t ) 0x1a; /* R26 - must be saved across function calls. Callee-save. */
256  pxTopOfStack--;
257  *pxTopOfStack = ( StackType_t ) 0x1b; /* R27 - must be saved across function calls. Callee-save. */
258  pxTopOfStack--;
259  *pxTopOfStack = ( StackType_t ) 0x1c; /* R28 - must be saved across function calls. Callee-save. */
260  pxTopOfStack--;
261  *pxTopOfStack = ( StackType_t ) 0x1d; /* R29 - must be saved across function calls. Callee-save. */
262  pxTopOfStack--;
263  *pxTopOfStack = ( StackType_t ) 0x1e; /* R30 - must be saved across function calls. Callee-save. */
264  pxTopOfStack--;
265  *pxTopOfStack = ( StackType_t ) 0x1f; /* R31 - must be saved across function calls. Callee-save. */
266  pxTopOfStack--;
267  #else
268  pxTopOfStack -= 13;
269  #endif
270 
271  /* Return a pointer to the top of the stack that has been generated so this
272  can be stored in the task control block for the task. */
273  return pxTopOfStack;
274 }

277 BaseType_t xPortStartScheduler( void )
278 {
279 extern void ( vPortStartFirstTask )( void );
280 extern uint32_t _stack[];
281 
282  /* Setup the hardware to generate the tick. Interrupts are disabled when
283  this function is called.
284 
285  This port uses an application defined callback function to install the tick
286  interrupt handler because the kernel will run on lots of different
287  MicroBlaze and FPGA configurations - not all of which will have the same
288  timer peripherals defined or available. An example definition of
289  vApplicationSetupTimerInterrupt() is provided in the official demo
290  application that accompanies this port. */
291  vApplicationSetupTimerInterrupt();
292 
293  /* Reuse the stack from main() as the stack for the interrupts/exceptions. */
294  pulISRStack = ( uint32_t * ) _stack;
295 
296  /* Ensure there is enough space for the functions called from the interrupt
297  service routines to write back into the stack frame of the caller. */
298  pulISRStack -= 2;
299 
300  /* Restore the context of the first task that is going to run. From here
301  on, the created tasks will be executing. */
302  vPortStartFirstTask();
303 
304  /* Should not get here as the tasks are now running! */
305  return pdFALSE;
306 }

317 /*
318  * Manual context switch called by portYIELD or taskYIELD.
319  */
320 void vPortYield( void )
321 {
322 extern void VPortYieldASM( void );
323 
324  /* Perform the context switch in a critical section to assure it is
325  not interrupted by the tick ISR. It is not a problem to do this as
326  each task maintains its own interrupt status. */
327  portENTER_CRITICAL();
328  {
329  /* Jump directly to the yield function to ensure there is no
330  compiler generated prologue code. */
331  asm volatile ( "bralid r14, VPortYieldASM \n\t" \
332  "or r0, r0, r0 \n\t" );
333  }
334  portEXIT_CRITICAL();
335 }

451 static int32_t prvInitialiseInterruptController( void )
452 {
453 int32_t lStatus;
454 
455  lStatus = XIntc_Initialize( &xInterruptControllerInstance, configINTERRUPT_CONTROLLER_TO_USE );
456 
457  if( lStatus == XST_SUCCESS )
458  {
459  /* Initialise the exception table. */
460  Xil_ExceptionInit();
461 
462  /* Service all pending interrupts each time the handler is entered. */
463  XIntc_SetIntrSvcOption( xInterruptControllerInstance.BaseAddress, XIN_SVC_ALL_ISRS_OPTION );
464 
465  /* Install exception handlers if the MicroBlaze is configured to handle
466  exceptions, and the application defined constant
467  configINSTALL_EXCEPTION_HANDLERS is set to 1. */
468  #if ( MICROBLAZE_EXCEPTIONS_ENABLED == 1 ) && ( configINSTALL_EXCEPTION_HANDLERS == 1 )
469  {
470  vPortExceptionsInstallHandlers();
471  }
472  #endif /* MICROBLAZE_EXCEPTIONS_ENABLED */
473 
474  /* Start the interrupt controller. Interrupts are enabled when the
475  scheduler starts. */
476  lStatus = XIntc_Start( &xInterruptControllerInstance, XIN_REAL_MODE );
477 
478  if( lStatus == XST_SUCCESS )
479  {
480  lStatus = pdPASS;
481  }
482  else
483  {
484  lStatus = pdFAIL;
485  }
486  }
487 
488  configASSERT( lStatus == pdPASS );
489 
490  return lStatus;
491 }
