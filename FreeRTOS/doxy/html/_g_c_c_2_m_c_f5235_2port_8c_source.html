104 StackType_t *
105 pxPortInitialiseStack( StackType_t * pxTopOfStack, TaskFunction_t pxCode,
106  void *pvParameters )
107 {
108  /* Place the parameter on the stack in the expected location. */
109  *pxTopOfStack = ( StackType_t ) pvParameters;
110  pxTopOfStack--;
111 
112  /* Place dummy return address on stack. Tasks should never terminate so
113  * we can set this to anything. */
114  *pxTopOfStack = ( StackType_t ) 0;
115  pxTopOfStack--;
116 
117  /* Create a Motorola Coldfire exception stack frame. First comes the return
118  * address. */
119  *pxTopOfStack = ( StackType_t ) pxCode;
120  pxTopOfStack--;
121 
122  /* Format, fault-status, vector number for exception stack frame. Task
123  * run in supervisor mode. */
124  *pxTopOfStack = 0x40002000UL | ( portVECTOR_SYSCALL + 32 ) << 18;
125  pxTopOfStack--;
126 
127  /* Set the initial critical section nesting counter to zero. This value
128  * is used to restore the value of ulCriticalNesting. */
129  *pxTopOfStack = 0;
130  *pxTopOfStack--;
131 
132  *pxTopOfStack = ( StackType_t ) 0xA6; /* A6 / FP */
133  pxTopOfStack--;
134  *pxTopOfStack = ( StackType_t ) 0xA5; /* A5 */
135  pxTopOfStack--;
136  *pxTopOfStack = ( StackType_t ) 0xA4; /* A4 */
137  pxTopOfStack--;
138  *pxTopOfStack = ( StackType_t ) 0xA3; /* A3 */
139  pxTopOfStack--;
140  *pxTopOfStack = ( StackType_t ) 0xA2; /* A2 */
141  pxTopOfStack--;
142  *pxTopOfStack = ( StackType_t ) 0xA1; /* A1 */
143  pxTopOfStack--;
144  *pxTopOfStack = ( StackType_t ) 0xA0; /* A0 */
145  pxTopOfStack--;
146  *pxTopOfStack = ( StackType_t ) 0xD7; /* D7 */
147  pxTopOfStack--;
148  *pxTopOfStack = ( StackType_t ) 0xD6; /* D6 */
149  pxTopOfStack--;
150  *pxTopOfStack = ( StackType_t ) 0xD5; /* D5 */
151  pxTopOfStack--;
152  *pxTopOfStack = ( StackType_t ) 0xD4; /* D4 */
153  pxTopOfStack--;
154  *pxTopOfStack = ( StackType_t ) 0xD3; /* D3 */
155  pxTopOfStack--;
156  *pxTopOfStack = ( StackType_t ) 0xD2; /* D2 */
157  pxTopOfStack--;
158  *pxTopOfStack = ( StackType_t ) 0xD1; /* D1 */
159  pxTopOfStack--;
160  *pxTopOfStack = ( StackType_t ) 0xD0; /* D0 */
161 
162  return pxTopOfStack;
163 }

165 /*
166  * Called by portYIELD() or taskYIELD() to manually force a context switch.
167  */
168 static void
169 prvPortYield( void )
170 {
171  asm volatile ( "move.w #0x2700, %sr\n\t" );
172 #if _GCC_USES_FP == 1
173  asm volatile ( "unlk %fp\n\t" );
174 #endif
175  /* Perform the context switch. First save the context of the current task. */
176  portSAVE_CONTEXT( );
177 
178  /* Find the highest priority task that is ready to run. */
179  vTaskSwitchContext( );
180 
181  /* Restore the context of the new task. */
182  portRESTORE_CONTEXT( );
183 }

220 void
221 vPortEnterCritical()
222 {
223  /* FIXME: We should store the old IPL here - How are we supposed to do
224  * this.
225  */
226  ( void )portSET_IPL( portIPL_MAX );
227 
228  /* Now interrupts are disabled ulCriticalNesting can be accessed
229  * directly. Increment ulCriticalNesting to keep a count of how many times
230  * portENTER_CRITICAL() has been called. */
231  ulCriticalNesting++;
232 }

234 void
235 vPortExitCritical()
236 {
237  if( ulCriticalNesting > portNO_CRITICAL_NESTING )
238  {
239  /* Decrement the nesting count as we are leaving a critical section. */
240  ulCriticalNesting--;
241 
242  /* If the nesting level has reached zero then interrupts should be
243  re-enabled. */
244  if( ulCriticalNesting == portNO_CRITICAL_NESTING )
245  {
246  ( void )portSET_IPL( 0 );
247  }
248  }
249 }

251 BaseType_t
252 xPortStartScheduler( void )
253 {
254  extern void ( *portVECTOR_TABLE[ ] ) ( );
255 
256  /* Add entry in vector table for yield system call. */
257  portVECTOR_TABLE[ portVECTOR_SYSCALL ] = prvPortYield;
258  /* Add entry in vector table for periodic timer. */
259  portVECTOR_TABLE[ portVECTOR_TIMER ] = prvPortPreemptiveTick;
260 
261  /* Configure the timer for the system clock. */
262  if ( configTICK_RATE_HZ > 0)
263  {
264  /* Configure prescaler */
265  MCF_PIT_PCSR0 = MCF_PIT_PCSR_PRE( 0x9 ) | MCF_PIT_PCSR_RLD | MCF_PIT_PCSR_OVW;
266  /* Initialize the periodic timer interrupt. */
267  MCF_PIT_PMR0 = MCF_PIT_MODULUS_REGISTER( configTICK_RATE_HZ );
268  /* Configure interrupt priority and level and unmask interrupt. */
269  MCF_INTC0_ICR36 = MCF_INTC0_ICRn_IL( 0x1 ) | MCF_INTC0_ICRn_IP( 0x1 );
270  MCF_INTC0_IMRH &= ~( MCF_INTC0_IMRH_INT_MASK36 | MCF_INTC0_IMRH_MASKALL );
271  /* Enable interrupts */
272  MCF_PIT_PCSR0 |= MCF_PIT_PCSR_PIE | MCF_PIT_PCSR_EN | MCF_PIT_PCSR_PIF;
273  }
274 
275  /* Restore the context of the first task that is going to run. */
276  portRESTORE_CONTEXT( );
277 
278  /* Should not get here. */
279  return pdTRUE;
280 }
