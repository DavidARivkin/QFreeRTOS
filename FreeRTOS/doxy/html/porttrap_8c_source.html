209 void vInternalProtectionTrap( int iTrapIdentification )
210 {
211  /* Deliberate fall through to default. */
212  switch( iTrapIdentification )
213  {
214  case portTIN_IPT_PRIVILIGED_INSTRUCTION:
215  /* Instruction is not allowed at current execution level, eg DISABLE at User-0. */
216 
217  case portTIN_IPT_MEMORY_PROTECTION_READ:
218  /* Load word using invalid address. */
219 
220  case portTIN_IPT_MEMORY_PROTECTION_WRITE:
221  /* Store Word using invalid address. */
222 
223  case portTIN_IPT_MEMORY_PROTECTION_EXECUTION:
224  /* PC jumped to an address outside of the valid range. */
225 
226  case portTIN_IPT_MEMORY_PROTECTION_PERIPHERAL_ACCESS:
227  /* Access to a peripheral denied at current execution level. */
228 
229  case portTIN_IPT_MEMORY_PROTECTION_NULL_ADDRESS:
230  /* NULL Pointer. */
231 
232  case portTIN_IPT_MEMORY_PROTECTION_GLOBAL_REGISTER_WRITE_PROTECTION:
233  /* Tried to modify a global address pointer register. */
234 
235  default:
236 
237  pxCurrentTCB[ 0 ] = _mfcr( $PCXI );
238  _debug();
239  break;
240  }
241 }
