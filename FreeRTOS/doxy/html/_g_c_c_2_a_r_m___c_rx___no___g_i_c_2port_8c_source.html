165 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters )
166 {
167  /* Setup the initial stack of the task. The stack is set exactly as
168  expected by the portRESTORE_CONTEXT() macro.
169 
170  The fist real value on the stack is the status register, which is set for
171  system mode, with interrupts enabled. A few NULLs are added first to ensure
172  GDB does not try decoding a non-existent return address. */
173  *pxTopOfStack = ( StackType_t ) NULL;
174  pxTopOfStack--;
175  *pxTopOfStack = ( StackType_t ) NULL;
176  pxTopOfStack--;
177  *pxTopOfStack = ( StackType_t ) NULL;
178  pxTopOfStack--;
179  *pxTopOfStack = ( StackType_t ) portINITIAL_SPSR;
180 
181  if( ( ( uint32_t ) pxCode & portTHUMB_MODE_ADDRESS ) != 0x00UL )
182  {
183  /* The task will start in THUMB mode. */
184  *pxTopOfStack |= portTHUMB_MODE_BIT;
185  }
186 
187  pxTopOfStack--;
188 
189  /* Next the return address, which in this case is the start of the task. */
190  *pxTopOfStack = ( StackType_t ) pxCode;
191  pxTopOfStack--;
192 
193  /* Next all the registers other than the stack pointer. */
194  *pxTopOfStack = ( StackType_t ) portTASK_RETURN_ADDRESS; /* R14 */
195  pxTopOfStack--;
196  *pxTopOfStack = ( StackType_t ) 0x12121212; /* R12 */
197  pxTopOfStack--;
198  *pxTopOfStack = ( StackType_t ) 0x11111111; /* R11 */
199  pxTopOfStack--;
200  *pxTopOfStack = ( StackType_t ) 0x10101010; /* R10 */
201  pxTopOfStack--;
202  *pxTopOfStack = ( StackType_t ) 0x09090909; /* R9 */
203  pxTopOfStack--;
204  *pxTopOfStack = ( StackType_t ) 0x08080808; /* R8 */
205  pxTopOfStack--;
206  *pxTopOfStack = ( StackType_t ) 0x07070707; /* R7 */
207  pxTopOfStack--;
208  *pxTopOfStack = ( StackType_t ) 0x06060606; /* R6 */
209  pxTopOfStack--;
210  *pxTopOfStack = ( StackType_t ) 0x05050505; /* R5 */
211  pxTopOfStack--;
212  *pxTopOfStack = ( StackType_t ) 0x04040404; /* R4 */
213  pxTopOfStack--;
214  *pxTopOfStack = ( StackType_t ) 0x03030303; /* R3 */
215  pxTopOfStack--;
216  *pxTopOfStack = ( StackType_t ) 0x02020202; /* R2 */
217  pxTopOfStack--;
218  *pxTopOfStack = ( StackType_t ) 0x01010101; /* R1 */
219  pxTopOfStack--;
220  *pxTopOfStack = ( StackType_t ) pvParameters; /* R0 */
221  pxTopOfStack--;
222 
223  /* The task will start with a critical nesting count of 0 as interrupts are
224  enabled. */
225  *pxTopOfStack = portNO_CRITICAL_NESTING;
226  pxTopOfStack--;
227 
228  /* The task will start without a floating point context. A task that uses
229  the floating point hardware must call vPortTaskUsesFPU() before executing
230  any floating point instructions. */
231  *pxTopOfStack = portNO_FLOATING_POINT_CONTEXT;
232 
233  return pxTopOfStack;
234 }

237 static void prvTaskExitError( void )
238 {
239  /* A function that implements a task must not exit or attempt to return to
240  its caller as there is nothing to return to. If a task wants to exit it
241  should instead call vTaskDelete( NULL ).
242 
243  Artificially force an assert() to be triggered if configASSERT() is
244  defined, then stop here so application writers can catch the error. */
245  configASSERT( ulPortInterruptNesting == ~0UL );
246  portDISABLE_INTERRUPTS();
247  for( ;; );
248 }

251 BaseType_t xPortStartScheduler( void )
252 {
253 uint32_t ulAPSR;
254 
255  /* Only continue if the CPU is not in User mode. The CPU must be in a
256  Privileged mode for the scheduler to start. */
257  __asm volatile ( "MRS %0, APSR" : "=r" ( ulAPSR ) );
258  ulAPSR &= portAPSR_MODE_BITS_MASK;
259  configASSERT( ulAPSR != portAPSR_USER_MODE );
260 
261  if( ulAPSR != portAPSR_USER_MODE )
262  {
263  /* Start the timer that generates the tick ISR. */
264  portDISABLE_INTERRUPTS();
265  configSETUP_TICK_INTERRUPT();
266 
267  /* Start the first task executing. */
268  vPortRestoreTaskContext();
269  }
270 
271  /* Will only get here if xTaskStartScheduler() was called with the CPU in
272  a non-privileged mode or the binary point register was not set to its lowest
273  possible value. prvTaskExitError() is referenced to prevent a compiler
274  warning about it being defined but not referenced in the case that the user
275  defines their own exit address. */
276  ( void ) prvTaskExitError;
277  return 0;
278 }

289 void vPortEnterCritical( void )
290 {
291  portDISABLE_INTERRUPTS();
292 
293  /* Now interrupts are disabled ulCriticalNesting can be accessed
294  directly. Increment ulCriticalNesting to keep a count of how many times
295  portENTER_CRITICAL() has been called. */
296  ulCriticalNesting++;
297 
298  /* This is not the interrupt safe version of the enter critical function so
299  assert() if it is being called from an interrupt context. Only API
300  functions that end in "FromISR" can be used in an interrupt. Only assert if
301  the critical nesting count is 1 to protect against recursive calls if the
302  assert function also uses a critical section. */
303  if( ulCriticalNesting == 1 )
304  {
305  configASSERT( ulPortInterruptNesting == 0 );
306  }
307 }

310 void vPortExitCritical( void )
311 {
312  if( ulCriticalNesting > portNO_CRITICAL_NESTING )
313  {
314  /* Decrement the nesting count as the critical section is being
315  exited. */
316  ulCriticalNesting--;
317 
318  /* If the nesting level has reached zero then all interrupt
319  priorities must be re-enabled. */
320  if( ulCriticalNesting == portNO_CRITICAL_NESTING )
321  {
322  /* Critical nesting has reached zero so all interrupt priorities
323  should be unmasked. */
324  portENABLE_INTERRUPTS();
325  }
326  }
327 }

334  ulInterruptStatus = portSET_INTERRUPT_MASK_FROM_ISR();

341

343

348 void vPortTaskUsesFPU( void )
349 {
350 uint32_t ulInitialFPSCR = 0;
351 
352  /* A task is registering the fact that it needs an FPU context. Set the
353  FPU flag (which is saved as part of the task context). */
354  ulPortTaskHasFPUContext = pdTRUE;
355 
356  /* Initialise the floating point status register. */
357  __asm volatile ( "FMXR FPSCR, %0" :: "r" (ulInitialFPSCR) );
358 }
