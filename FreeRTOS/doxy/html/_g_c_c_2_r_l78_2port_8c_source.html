70 /* The critical nesting value is initialised to a non zero value to ensure
71 interrupts don't accidentally become enabled before the scheduler is started. */
72 #define portINITIAL_CRITICAL_NESTING ( ( uint16_t ) 10 )
73 
74 /* Initial PSW value allocated to a newly created task.
75  * 11000110
76  * ||||||||-------------- Fill byte
77  * |||||||--------------- Carry Flag cleared
78  * |||||----------------- In-service priority Flags set to low level
79  * ||||------------------ Register bank Select 0 Flag cleared
80  * |||------------------- Auxiliary Carry Flag cleared
81  * ||-------------------- Register bank Select 1 Flag cleared
82  * |--------------------- Zero Flag set
83  * ---------------------- Global Interrupt Flag set (enabled)
84  */
85 #define portPSW ( 0xc6UL )
86 
87 /* Each task maintains a count of the critical section nesting depth. Each time
88 a critical section is entered the count is incremented. Each time a critical
89 section is exited the count is decremented - with interrupts only being
90 re-enabled if the count is zero.
91 
92 usCriticalNesting will get set to zero when the scheduler starts, but must
93 not be initialised to zero as that could cause problems during the startup
94 sequence. */
95 volatile uint16_t usCriticalNesting = portINITIAL_CRITICAL_NESTING;

112 /*
113  * Initialise the stack of a task to look exactly as if a call to
114  * portSAVE_CONTEXT had been called.
115  *
116  * See the header file portable.h.
117  */
118 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters )
119 {
120 uint32_t *pulLocal;
121 
122  /* Stack type and pointers to the stack type are both 2 bytes. */
123 
124  /* Parameters are passed in on the stack, and written using a 32bit value
125  hence a space is left for the second two bytes. */
126  pxTopOfStack--;
127 
128  /* Write in the parameter value. */
129  pulLocal = ( uint32_t * ) pxTopOfStack;
130  *pulLocal = ( StackType_t ) pvParameters;
131  pxTopOfStack--;
132 
133  /* The return address, leaving space for the first two bytes of the
134  32-bit value. */
135  pxTopOfStack--;
136  pulLocal = ( uint32_t * ) pxTopOfStack;
137  *pulLocal = ( uint32_t ) 0;
138  pxTopOfStack--;
139 
140  /* The start address / PSW value is also written in as a 32bit value,
141  so leave a space for the second two bytes. */
142  pxTopOfStack--;
143 
144  /* Task function start address combined with the PSW. */
145  pulLocal = ( uint32_t * ) pxTopOfStack;
146  *pulLocal = ( ( ( uint32_t ) pxCode ) | ( portPSW << 24UL ) );
147  pxTopOfStack--;
148 
149  /* An initial value for the AX register. */
150  *pxTopOfStack = ( StackType_t ) 0x1111;
151  pxTopOfStack--;
152 
153  /* An initial value for the HL register. */
154  *pxTopOfStack = ( StackType_t ) 0x2222;
155  pxTopOfStack--;
156 
157  /* CS and ES registers. */
158  *pxTopOfStack = ( StackType_t ) 0x0F00;
159  pxTopOfStack--;
160 
161  /* The remaining general purpose registers bank 0 (DE and BC) and the other
162  two register banks...register bank 3 is dedicated for use by interrupts so
163  is not saved as part of the task context. */
164  pxTopOfStack -= 10;
165 
166  /* Finally the critical section nesting count is set to zero when the task
167  first starts. */
168  *pxTopOfStack = ( StackType_t ) portNO_CRITICAL_SECTION_NESTING;
169 
170  /* Return a pointer to the top of the stack that has beene generated so it
171  can be stored in the task control block for the task. */
172  return pxTopOfStack;
173 }

196 __attribute__((weak)) void vApplicationSetupTimerInterrupt( void )
197 {
198 const uint16_t usClockHz = 15000UL; /* Internal clock. */
199 const uint16_t usCompareMatch = ( usClockHz / configTICK_RATE_HZ ) + 1UL;
200 
201  /* Use the internal 15K clock. */
202  OSMC = ( unsigned char ) 0x16;
203 
204  #ifdef RTCEN
205  {
206  /* Supply the interval timer clock. */
207  RTCEN = ( unsigned char ) 1U;
208 
209  /* Disable INTIT interrupt. */
210  ITMK = ( unsigned char ) 1;
211 
212  /* Disable ITMC operation. */
213  ITMC = ( unsigned char ) 0x0000;
214 
215  /* Clear INIT interrupt. */
216  ITIF = ( unsigned char ) 0;
217 
218  /* Set interval and enable interrupt operation. */
219  ITMC = usCompareMatch | 0x8000U;
220 
221  /* Enable INTIT interrupt. */
222  ITMK = ( unsigned char ) 0;
223  }
224  #endif
225 
226  #ifdef TMKAEN
227  {
228  /* Supply the interval timer clock. */
229  TMKAEN = ( unsigned char ) 1U;
230 
231  /* Disable INTIT interrupt. */
232  TMKAMK = ( unsigned char ) 1;
233 
234  /* Disable ITMC operation. */
235  ITMC = ( unsigned char ) 0x0000;
236 
237  /* Clear INIT interrupt. */

239 
240  /* Set interval and enable interrupt operation. */

247 }
