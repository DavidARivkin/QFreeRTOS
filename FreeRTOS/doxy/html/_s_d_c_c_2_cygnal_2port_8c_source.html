119 /*-----------------------------------------------------------*/
120 /*
121  * Macro that copies the current stack from internal RAM to XRAM. This is
122  * required as the 8051 only contains enough internal RAM for a single stack,
123  * but we have a stack for every task.
124  */
125 #define portCOPY_STACK_TO_XRAM() \
126 { \
127  /* pxCurrentTCB points to a TCB which itself points to the location into \
128  which the first stack byte should be copied. Set pxXRAMStack to point \
129  to the location into which the first stack byte is to be copied. */ \
130  pxXRAMStack = ( xdata StackType_t * ) *( ( xdata StackType_t ** ) pxCurrentTCB ); \
131  \
132  /* Set pxRAMStack to point to the first byte to be coped from the stack. */ \
133  pxRAMStack = ( data StackType_t * data ) configSTACK_START; \
134  \
135  /* Calculate the size of the stack we are about to copy from the current \
136  stack pointer value. */ \
137  ucStackBytes = SP - ( configSTACK_START - 1 ); \
138  \
139  /* Before starting to copy the stack, store the calculated stack size so \
140  the stack can be restored when the task is resumed. */ \
141  *pxXRAMStack = ucStackBytes; \
142  \
143  /* Copy each stack byte in turn. pxXRAMStack is incremented first as we \
144  have already stored the stack size into XRAM. */ \
145  while( ucStackBytes ) \
146  { \
147  pxXRAMStack++; \
148  *pxXRAMStack = *pxRAMStack; \
149  pxRAMStack++; \
150  ucStackBytes--; \
151  } \
152 }
153 /*-----------------------------------------------------------*/
154 
155 /*
156  * Macro that copies the stack of the task being resumed from XRAM into
157  * internal RAM.
158  */
159 #define portCOPY_XRAM_TO_STACK() \
160 { \
161  /* Setup the pointers as per portCOPY_STACK_TO_XRAM(), but this time to \
162  copy the data back out of XRAM and into the stack. */ \
163  pxXRAMStack = ( xdata StackType_t * ) *( ( xdata StackType_t ** ) pxCurrentTCB ); \
164  pxRAMStack = ( data StackType_t * data ) ( configSTACK_START - 1 ); \
165  \
166  /* The first value stored in XRAM was the size of the stack - i.e. the \
167  number of bytes we need to copy back. */ \
168  ucStackBytes = pxXRAMStack[ 0 ]; \
169  \
170  /* Copy the required number of bytes back into the stack. */ \
171  do \
172  { \
173  pxXRAMStack++; \
174  pxRAMStack++; \
175  *pxRAMStack = *pxXRAMStack; \
176  ucStackBytes--; \
177  } while( ucStackBytes ); \
178  \
179  /* Restore the stack pointer ready to use the restored stack. */ \
180  SP = ( uint8_t ) pxRAMStack; \
181 }

184 /*
185  * Macro to push the current execution context onto the stack, before the stack
186  * is moved to XRAM.
187  */
188 #define portSAVE_CONTEXT() \
189 { \
190  _asm \
191  /* Push ACC first, as when restoring the context it must be restored \
192  last (it is used to set the IE register). */ \
193  push ACC \
194  /* Store the IE register then disable interrupts. */ \
195  push IE \
196  clr _EA \
197  push DPL \
198  push DPH \
199  push b \
200  push ar2 \
201  push ar3 \
202  push ar4 \
203  push ar5 \
204  push ar6 \
205  push ar7 \
206  push ar0 \
207  push ar1 \
208  push PSW \
209  _endasm; \
210  PSW = 0; \
211  _asm \
212  push _bp \
213  _endasm; \
214 }
215 /*-----------------------------------------------------------*/
216 
217 /*
218  * Macro that restores the execution context from the stack. The execution
219  * context was saved into the stack before the stack was copied into XRAM.
220  */
221 #define portRESTORE_CONTEXT() \
222 { \
223  _asm \
224  pop _bp \
225  pop PSW \
226  pop ar1 \
227  pop ar0 \
228  pop ar7 \
229  pop ar6 \
230  pop ar5 \
231  pop ar4 \
232  pop ar3 \
233  pop ar2 \
234  pop b \
235  pop DPH \
236  pop DPL \
237  /* The next byte of the stack is the IE register. Only the global \
238  enable bit forms part of the task context. Pop off the IE then set \
239  the global enable bit to match that of the stored IE register. */ \
240  pop ACC \
241  JB ACC.7,0098$ \
242  CLR IE.7 \
243  LJMP 0099$ \
244  0098$: \
245  SETB IE.7 \
246  0099$: \
247  /* Finally pop off the ACC, which was the first register saved. */ \
248  pop ACC \
249  reti \
250  _endasm; \
251 }

257 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters )
258 {
259 uint32_t ulAddress;
260 StackType_t *pxStartOfStack;
261 
262  /* Leave space to write the size of the stack as the first byte. */
263  pxStartOfStack = pxTopOfStack;
264  pxTopOfStack++;
265 
266  /* Place a few bytes of known values on the bottom of the stack.
267  This is just useful for debugging and can be uncommented if required.
268  *pxTopOfStack = 0x11;
269  pxTopOfStack++;
270  *pxTopOfStack = 0x22;
271  pxTopOfStack++;
272  *pxTopOfStack = 0x33;
273  pxTopOfStack++;
274  */
275 
276  /* Simulate how the stack would look after a call to the scheduler tick
277  ISR.
278 
279  The return address that would have been pushed by the MCU. */
280  ulAddress = ( uint32_t ) pxCode;
281  *pxTopOfStack = ( StackType_t ) ulAddress;
282  ulAddress >>= 8;
283  pxTopOfStack++;
284  *pxTopOfStack = ( StackType_t ) ( ulAddress );
285  pxTopOfStack++;
286 
287  /* Next all the registers will have been pushed by portSAVE_CONTEXT(). */
288  *pxTopOfStack = 0xaa; /* acc */
289  pxTopOfStack++;
290 
291  /* We want tasks to start with interrupts enabled. */
292  *pxTopOfStack = portGLOBAL_INTERRUPT_BIT;
293  pxTopOfStack++;
294 
295  /* The function parameters will be passed in the DPTR and B register as
296  a three byte generic pointer is used. */
297  ulAddress = ( uint32_t ) pvParameters;
298  *pxTopOfStack = ( StackType_t ) ulAddress; /* DPL */
299  ulAddress >>= 8;
300  *pxTopOfStack++;
301  *pxTopOfStack = ( StackType_t ) ulAddress; /* DPH */
302  ulAddress >>= 8;
303  pxTopOfStack++;
304  *pxTopOfStack = ( StackType_t ) ulAddress; /* b */
305  pxTopOfStack++;
306 
307  /* The remaining registers are straight forward. */
308  *pxTopOfStack = 0x02; /* R2 */
309  pxTopOfStack++;
310  *pxTopOfStack = 0x03; /* R3 */
311  pxTopOfStack++;
312  *pxTopOfStack = 0x04; /* R4 */
313  pxTopOfStack++;
314  *pxTopOfStack = 0x05; /* R5 */
315  pxTopOfStack++;
316  *pxTopOfStack = 0x06; /* R6 */
317  pxTopOfStack++;
318  *pxTopOfStack = 0x07; /* R7 */
319  pxTopOfStack++;
320  *pxTopOfStack = 0x00; /* R0 */
321  pxTopOfStack++;
322  *pxTopOfStack = 0x01; /* R1 */
323  pxTopOfStack++;
324  *pxTopOfStack = 0x00; /* PSW */
325  pxTopOfStack++;
326  *pxTopOfStack = 0xbb; /* BP */
327 
328  /* Dont increment the stack size here as we don't want to include
329  the stack size byte as part of the stack size count.
330 
331  Finally we place the stack size at the beginning. */
332  *pxStartOfStack = ( StackType_t ) ( pxTopOfStack - pxStartOfStack );
333 
334  /* Unlike most ports, we return the start of the stack as this is where the
335  size of the stack is stored. */
336  return pxStartOfStack;
337 }

343 BaseType_t xPortStartScheduler( void )
344 {
345  /* Setup timer 2 to generate the RTOS tick. */
346  prvSetupTimerInterrupt();

348  /* Make sure we start with the expected SFR page. This line should not

355  portRESTORE_CONTEXT();
356 
357  /* Should never get here! */
358  return pdTRUE;

368 /*
369  * Manual context switch. The first thing we do is save the registers so we
370  * can use a naked attribute.
371  */
372 void vPortYield( void ) _naked
373 {
374  /* Save the execution context onto the stack, then copy the entire stack
375  to XRAM. This is necessary as the internal RAM is only large enough to
376  hold one stack, and we want one per task.
377 
378  PERFORMANCE COULD BE IMPROVED BY ONLY COPYING TO XRAM IF A TASK SWITCH
379  IS REQUIRED. */
380  portSAVE_CONTEXT();
381  portCOPY_STACK_TO_XRAM();
382 
383  /* Call the standard scheduler context switch function. */
384  vTaskSwitchContext();
385 
386  /* Copy the stack of the task about to execute from XRAM into RAM and
387  restore it's context ready to run on exiting. */
388  portCOPY_XRAM_TO_STACK();
389  portRESTORE_CONTEXT();
390 }

424 static void prvSetupTimerInterrupt( void )
425 {
426 uint8_t ucOriginalSFRPage;
427 
428 /* Constants calculated to give the required timer capture values. */
429 const uint32_t ulTicksPerSecond = configCPU_CLOCK_HZ / portCLOCK_DIVISOR;
430 const uint32_t ulCaptureTime = ulTicksPerSecond / configTICK_RATE_HZ;
431 const uint32_t ulCaptureValue = portMAX_TIMER_VALUE - ulCaptureTime;
432 const uint8_t ucLowCaptureByte = ( uint8_t ) ( ulCaptureValue & ( uint32_t ) 0xff );
433 const uint8_t ucHighCaptureByte = ( uint8_t ) ( ulCaptureValue >> ( uint32_t ) 8 );
434 
435  /* NOTE: This uses a timer only present on 8052 architecture. */
436 
437  /* Remember the current SFR page so we can restore it at the end of the
438  function. */
439  ucOriginalSFRPage = SFRPAGE;
440  SFRPAGE = 0;
441 
442  /* TMR2CF can be left in its default state. */
443  TMR2CF = ( uint8_t ) 0;
444 
445  /* Setup the overflow reload value. */
446  RCAP2L = ucLowCaptureByte;
447  RCAP2H = ucHighCaptureByte;
448 
449  /* The initial load is performed manually. */
450  TMR2L = ucLowCaptureByte;
451  TMR2H = ucHighCaptureByte;
452 
453  /* Enable the timer 2 interrupts. */
454  IE |= portTIMER_2_INTERRUPT_ENABLE;
455 
456  /* Interrupts are disabled when this is called so the timer can be started
457  here. */
458  TMR2CN = portENABLE_TIMER;
459 
460  /* Restore the original SFR page. */
461  SFRPAGE = ucOriginalSFRPage;
462 }
