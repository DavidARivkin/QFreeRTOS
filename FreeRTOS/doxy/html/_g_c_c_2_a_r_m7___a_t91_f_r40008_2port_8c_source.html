112 /*
113  * Initialise the stack of a task to look exactly as if a call to
114  * portSAVE_CONTEXT had been called.
115  *
116  * See header file for description.
117  */
118 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters )
119 {
120 StackType_t *pxOriginalTOS;
121 
122  pxOriginalTOS = pxTopOfStack;
123 
124  /* To ensure asserts in tasks.c don't fail, although in this case the assert
125  is not really required. */
126  pxTopOfStack--;
127 
128  /* Setup the initial stack of the task. The stack is set exactly as
129  expected by the portRESTORE_CONTEXT() macro. */
130 
131  /* First on the stack is the return address - which in this case is the
132  start of the task. The offset is added to make the return address appear
133  as it would within an IRQ ISR. */
134  *pxTopOfStack = ( StackType_t ) pxCode + portINSTRUCTION_SIZE;
135  pxTopOfStack--;
136 
137  *pxTopOfStack = ( StackType_t ) 0xaaaaaaaa; /* R14 */
138  pxTopOfStack--;
139  *pxTopOfStack = ( StackType_t ) pxOriginalTOS; /* Stack used when task starts goes in R13. */
140  pxTopOfStack--;
141  *pxTopOfStack = ( StackType_t ) 0x12121212; /* R12 */
142  pxTopOfStack--;
143  *pxTopOfStack = ( StackType_t ) 0x11111111; /* R11 */
144  pxTopOfStack--;
145  *pxTopOfStack = ( StackType_t ) 0x10101010; /* R10 */
146  pxTopOfStack--;
147  *pxTopOfStack = ( StackType_t ) 0x09090909; /* R9 */
148  pxTopOfStack--;
149  *pxTopOfStack = ( StackType_t ) 0x08080808; /* R8 */
150  pxTopOfStack--;
151  *pxTopOfStack = ( StackType_t ) 0x07070707; /* R7 */
152  pxTopOfStack--;
153  *pxTopOfStack = ( StackType_t ) 0x06060606; /* R6 */
154  pxTopOfStack--;
155  *pxTopOfStack = ( StackType_t ) 0x05050505; /* R5 */
156  pxTopOfStack--;
157  *pxTopOfStack = ( StackType_t ) 0x04040404; /* R4 */
158  pxTopOfStack--;
159  *pxTopOfStack = ( StackType_t ) 0x03030303; /* R3 */
160  pxTopOfStack--;
161  *pxTopOfStack = ( StackType_t ) 0x02020202; /* R2 */
162  pxTopOfStack--;
163  *pxTopOfStack = ( StackType_t ) 0x01010101; /* R1 */
164  pxTopOfStack--;
165 
166  /* When the task starts is will expect to find the function parameter in
167  R0. */
168  *pxTopOfStack = ( StackType_t ) pvParameters; /* R0 */
169  pxTopOfStack--;
170 
171  /* The last thing onto the stack is the status register, which is set for
172  system mode, with interrupts enabled. */
173  *pxTopOfStack = ( StackType_t ) portINITIAL_SPSR;
174 
175  #ifdef THUMB_INTERWORK
176  {
177  /* We want the task to start in thumb mode. */
178  *pxTopOfStack |= portTHUMB_MODE_BIT;
179  }
180  #endif
181 
182  pxTopOfStack--;
183 
184  /* Some optimisation levels use the stack differently to others. This
185  means the interrupt flags cannot always be stored on the stack and will
186  instead be stored in a variable, which is then saved as part of the
187  tasks context. */
188  *pxTopOfStack = portNO_CRITICAL_SECTION_NESTING;
189 
190  return pxTopOfStack;
191 }

215 /*
216  * Setup the tick timer to generate the tick interrupts at the required frequency.
217  */
218 static void prvSetupTimerInterrupt( void )
219 {
220 volatile uint32_t ulDummy;
221 
222  /* Enable clock to the tick timer... */
223  AT91C_BASE_PS->PS_PCER = portTIMER_CLK_ENABLE_BIT;
224 
225  /* Stop the tick timer... */
226  portTIMER_REG_BASE_PTR->TC_CCR = TC_CLKDIS;
227 
228  /* Start with tick timer interrupts disabled... */
229  portTIMER_REG_BASE_PTR->TC_IDR = 0xFFFFFFFF;
230 
231  /* Clear any pending tick timer interrupts... */
232  ulDummy = portTIMER_REG_BASE_PTR->TC_SR;
233 
234  /* Store interrupt handler function address in tick timer vector register...
235  The ISR installed depends on whether the preemptive or cooperative
236  scheduler is being used. */
237  #if configUSE_PREEMPTION == 1
238  {
239  extern void ( vPreemptiveTick )( void );
240  AT91C_BASE_AIC->AIC_SVR[portTIMER_AIC_CHANNEL] = ( uint32_t ) vPreemptiveTick;
241  }
242  #else // else use cooperative scheduler
243  {
244  extern void ( vNonPreemptiveTick )( void );
245  AT91C_BASE_AIC->AIC_SVR[portTIMER_AIC_CHANNEL] = ( uint32_t ) vNonPreemptiveTick;
246  }
247  #endif
248 
249  /* Tick timer interrupt level-sensitive, priority 6... */
250  AT91C_BASE_AIC->AIC_SMR[ portTIMER_AIC_CHANNEL ] = AIC_SRCTYPE_INT_LEVEL_SENSITIVE | portTICK_PRIORITY_6;
251 
252  /* Enable the tick timer interrupt...
253 
254  First at timer level */
255  portTIMER_REG_BASE_PTR->TC_IER = TC_CPCS;
256 
257  /* Then at the AIC level. */
258  AT91C_BASE_AIC->AIC_IECR = (1 << portTIMER_AIC_CHANNEL);
259 
260  /* Calculate timer compare value to achieve the desired tick rate... */
261  if( (configCPU_CLOCK_HZ / (configTICK_RATE_HZ * 2) ) <= 0xFFFF )
262  {
263  /* The tick rate is fast enough for us to use the faster timer input
264  clock (main clock / 2). */
265  portTIMER_REG_BASE_PTR->TC_CMR = TC_WAVE | TC_CLKS_MCK2 | TC_BURST_NONE | TC_CPCTRG;
266  portTIMER_REG_BASE_PTR->TC_RC = configCPU_CLOCK_HZ / (configTICK_RATE_HZ * 2);
267  }
268  else
269  {
270  /* We must use a slower timer input clock (main clock / 8) because the
271  tick rate is too slow for the faster input clock. */
272  portTIMER_REG_BASE_PTR->TC_CMR = TC_WAVE | TC_CLKS_MCK8 | TC_BURST_NONE | TC_CPCTRG;
273  portTIMER_REG_BASE_PTR->TC_RC = configCPU_CLOCK_HZ / (configTICK_RATE_HZ * 8);
274  }
275 
276  /* Start tick timer... */
277  portTIMER_REG_BASE_PTR->TC_CCR = TC_SWTRG | TC_CLKEN;
278 }
